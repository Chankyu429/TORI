## Character.cs
> **기본 생성자, private로 만든걸 public로 작성하여 사용 가능**
```csharp
class Character
{
    private string name = "";
    public string Name => name;

    private float hp;
    public float HP => hp;

    public Character(string name, float hp)
    {
        this.name = name;
        this.hp = hp;
    }
}

class Player : Character
{
    private float attack;
    public float Attack => attack;

    public Player(string name, float hp, float attack)
        : base(name, hp)
    {
        this.attack = attack;
    }
}

class Monster : Character
{
    private float mana;
    public float Mana => mana;

    public Monster(string name, float hp, float mana)
        : base(name, hp)
    {
        this.mana = mana;
    }
}
```

### Character.cs
```csharp
class Character
{
    public string Name { get; protected set; }

    public int Hp { get; set; }

    public int Power { get; protected set; }
    
    public virtual void Attack(Character character)
    {
        string str = "";
        str += $"{Name}이 {character.Name}을 공격\n";

        if (character.Hp > 0)
            str += $"{character.Name}의 체력이 {character.Hp}만큼 남음\n";
        else
            str += $"{character.Name} 사망!!";
    }
}

class Monster : Character
{
    public Monster(string name, int hp, int power)
    {
        Name = name;
        Hp = hp;
        Power = power;
    }

    public override void Attack(Character character)
    {
        character.Hp -= Power;

        base.Attack(character);
    }
}

class Player : Character
{
    public int Up { get; protected set; }

    public Player(string name, int hp, int power, int up)
    {
        Name = name;
        Hp = hp;
        Power = power;
        Up = up;
    }

    public override void Attack(Character character)
    {
        character.Hp -= Power + Up;

        base.Attack(character);
    }
}
```
