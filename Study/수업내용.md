## Character.cs
> **기본 생성자, private로 만든걸 public로 작성하여 사용 가능**
```csharp
class Character
{
    private string name = "";
    public string Name => name;

    private float hp;
    public float HP => hp;

    public Character(string name, float hp)
    {
        this.name = name;
        this.hp = hp;
    }
}

class Player : Character
{
    private float attack;
    public float Attack => attack;

    public Player(string name, float hp, float attack)
        : base(name, hp)
    {
        this.attack = attack;
    }
}

class Monster : Character
{
    private float mana;
    public float Mana => mana;

    public Monster(string name, float hp, float mana)
        : base(name, hp)
    {
        this.mana = mana;
    }
}
```

### Character.cs
```csharp
class Character
{
    public string Name { get; protected set; }

    public int Hp { get; set; }

    public int Power { get; protected set; }
    
    public virtual void Attack(Character character)
    {
        string str = "";
        str += $"{Name}이 {character.Name}을 공격\n";

        if (character.Hp > 0)
            str += $"{character.Name}의 체력이 {character.Hp}만큼 남음\n";
        else
            str += $"{character.Name} 사망!!";
    }
}

class Monster : Character
{
    public Monster(string name, int hp, int power)
    {
        Name = name;
        Hp = hp;
        Power = power;
    }

    public override void Attack(Character character)
    {
        character.Hp -= Power;

        base.Attack(character);
    }
}

class Player : Character
{
    public int Up { get; protected set; }

    public Player(string name, int hp, int power, int up)
    {
        Name = name;
        Hp = hp;
        Power = power;
        Up = up;
    }

    public override void Attack(Character character)
    {
        character.Hp -= Power + Up;

        base.Attack(character);
    }
}
```

#### 오늘 사용한 코드
```csharp
for(int i = 0; i < random.Next(5, 10 + 1); i++)
{
    int[] nameNumbers = new int[3];
    nameNumbers[0] = random.Next(0, 14); //0 ~ 13
    nameNumbers[1] = random.Next(0, 14); //0 ~ 13
    nameNumbers[2] = random.Next(0, 14); //0 ~ 13

    
    string monsterName = "";
    monsterName += str[nameNumbers[0]];
    monsterName += str[nameNumbers[1]];
    monsterName += str[nameNumbers[2]];

    
    hp = random.Next(10, 50 + 1);
    power = random.Next(10, 50 + 1);

    monsterList.Add(new Monster(monsterName, hp, power));
}

Console.WriteLine();

for (int i = 0; i < monsterList.Count; i++)
{
    Monster m = monsterList[i];
    Console.WriteLine($"{i}번 이름 : {m.Name}, 체력 : {m.Hp}, 공격력 : {m.Power}");
}

Console.WriteLine();

while(true)
{
    Console.Write("공격할 몬스터 번호 : ");
    string input = Console.ReadLine()!;

    int monsterNumber;
    int.TryParse(input, out monsterNumber);

    if (monsterNumber < 0 || monsterNumber >= monsterList.Count)
    {
        Console.WriteLine("몬스터 번호를 잘못 입력했습니다!");

        continue;
    }


    player.Attack(monsterList[monsterNumber]);

    monsterNumber = random.Next(0, 10 + monsterList.Count + 1);
    if (monsterNumber >= monsterList.Count)
        continue;

    monsterList[monsterNumber].Attack(player);

    if(player.Hp < 0)
    {
        Console.WriteLine("플레이어 사망!");
        Console.WriteLine("Game Over!");

        break;
    }
}
```
